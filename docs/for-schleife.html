Code-Beispiel 
function updateFeedback(studentData) {
  const weakTopics = studentData.answers
    .filter(answer => !answer.correct)
    .map(answer => answer.topic);

  const priorities = [...new Set(weakTopics)].slice(0, 3);

  return {
    diagnosis: `Achte besonders auf: ${priorities.join(', ')}`,
    nextStep: 'Bearbeite zuerst eine Aufgabe mit mittlerem Niveau.',
    reflection: 'Kannst du deinen Lösungsweg in 2 Sätzen erklären?'
  };
}
